*** Cấp phép (provisioning) to the Host
docker-machine create \         
--driver digitalocean \
--digitalocean-image ubuntu-18-04-x64 \
--digitalocean-access-token {token}\
--engine-install-url "https://releases.rancher.com/install-docker/19.03.9.sh" \
{app name}
